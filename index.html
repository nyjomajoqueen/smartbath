#include #include #include #include #include #include // Настройки сети и базы #define WIFI_SSID "dstmobile" #define WIFI_PASSWORD "mobile2148" #define FIREBASE_HOST "smartbath-7cde1-default-rtdb.firebaseio.com" // Пины #define WATER_SENSOR_PIN D2 #define ONE_WIRE_BUS D3 #define LED_GREEN D4 #define LED_RED D5 OneWire oneWire(ONE_WIRE_BUS); DallasTemperature sensors(&oneWire); // Объекты Firebase FirebaseData fbdo; FirebaseData stream; FirebaseAuth auth; FirebaseConfig config; String systemState = "idle"; float lastTemp = -999.0; unsigned long lastTempTime = 0; volatile bool sensorChanged = false; // Прерывание на изменение состояния воды void IRAM_ATTR waterISR() { sensorChanged = true; } // Отдельная функция для приема данных из базы void streamCallback(StreamData data) { if (data.dataType() == "string") { systemState = data.stringData(); } } void streamTimeoutCallback(bool timeout) {} void sendToFirebase(String path, String val) { WiFiClientSecure client; client.setInsecure(); HTTPClient http; if (http.begin(client, "https://" + String(FIREBASE_HOST) + path)) { http.PUT(val); http.end(); } } void setup() { Serial.begin(115200); pinMode(WATER_SENSOR_PIN, INPUT_PULLUP); pinMode(LED_GREEN, OUTPUT); pinMode(LED_RED, OUTPUT); sensors.begin(); digitalWrite(LED_GREEN, HIGH); digitalWrite(LED_RED, LOW); WiFi.begin(WIFI_SSID, WIFI_PASSWORD); while (WiFi.status() != WL_CONNECTED) { delay(500); } // Настройка Firebase config.database_url = FIREBASE_HOST; config.signer.test_mode = true; Firebase.begin(&config, &auth); Firebase.reconnectWiFi(true); fbdo.setBSSLBufferSize(1024, 1024); stream.setBSSLBufferSize(1024, 1024); // Слушаем статус из базы Firebase.setStreamCallback(stream, streamCallback, streamTimeoutCallback); Firebase.beginStream(stream, "/system_state"); // ПО УМОЛЧАНИЮ: Проверяем датчик при старте и задаем статус if (digitalRead(WATER_SENSOR_PIN) == HIGH) { systemState = "filled"; } else { systemState = "idle"; } sendToFirebase("/system_state.json", "\"" + systemState + "\""); attachInterrupt(digitalPinToInterrupt(WATER_SENSOR_PIN), waterISR, CHANGE); } void loop() { ESP.wdtFeed(); yield(); // 1. ОБРАБОТКА ВОДЫ (Прерывание) if (sensorChanged) { sensorChanged = false; delay(100); // антидребезг bool isWet = (digitalRead(WATER_SENSOR_PIN) == HIGH); if (isWet && systemState == "idle") { systemState = "filled"; Firebase.setString(fbdo, "/system_state", "filled"); } else if (!isWet && systemState == "filled") { systemState = "idle"; Firebase.setString(fbdo, "/system_state", "idle"); } } // 2. ЛОГИКА СВЕТОДИОДОВ (Зависит от статуса) if (systemState == "idle") { digitalWrite(LED_GREEN, HIGH); digitalWrite(LED_RED, LOW); } else { digitalWrite(LED_GREEN, LOW); digitalWrite(LED_RED, HIGH); } // 3. АБСОЛЮТНО НЕЗАВИСИМОЕ ИЗМЕРЕНИЕ ТЕМПЕРАТУРЫ (Работает всегда!) if (millis() - lastTempTime > 2000) { lastTempTime = millis(); sensors.requestTemperatures(); float t = sensors.getTempCByIndex(0); if (t != DEVICE_DISCONNECTED_C) { // Ручной надежный расчет разницы float diff = t - lastTemp; if (diff < 0) { diff = -diff; } // Отправляем температуру, если это первый запуск или разница >= 0.3 if (lastTemp == -999.0 || diff >= 0.3) { lastTemp = t; Firebase.setFloat(fbdo, "/temperature", t); Serial.print("Температура обновлена: "); Serial.println(t); } } } }
